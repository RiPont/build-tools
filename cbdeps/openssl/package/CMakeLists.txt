IF (WIN32)
  SET (OPENSSL_UNPACKED_ROOT "./Program Files/Couchbase/Server")
ELSE ()
  SET (OPENSSL_UNPACKED_ROOT ./opt/couchbase)
ENDIF ()

FILE (COPY ${OPENSSL_UNPACKED_ROOT}/bin DESTINATION "${CMAKE_INSTALL_PREFIX}")
FILE (COPY ${OPENSSL_UNPACKED_ROOT}/lib DESTINATION "${CMAKE_INSTALL_PREFIX}")

# Our openssl 1.x builds do not have an etc dir
IF (EXISTS ${OPENSSL_UNPACKED_ROOT}/etc)
  FILE (COPY ${OPENSSL_UNPACKED_ROOT}/etc DESTINATION "${CMAKE_INSTALL_PREFIX}")
ENDIF ()

IF (WIN32)
  # Don't want to ship the static libs
  FILE (REMOVE_RECURSE "${CMAKE_INSTALL_PREFIX}/lib/VC")
ENDIF ()
SET_PROPERTY (GLOBAL APPEND PROPERTY CBDEPS_PREFIX_PATH "${CMAKE_CURRENT_SOURCE_DIR}")
